var popup = new mapboxgl.Popup({ closeButton: false, closeOnClick: false }); map.on('mouseenter', 'points', function(e) { // Change the cursor style as a UI indicator. map.getCanvas().style.cursor = 'pointer'; // Populate the popup and set its coordinates // based on the feature found. popup.setLngLat(e.features[0].geometry.coordinates) .setHTML(e.features[0].properties.description) .addTo(map); }); map.on('mouseleave', 'points', function() { map.getCanvas().style.cursor = ''; popup.remove(); });
